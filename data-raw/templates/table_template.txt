#' The "{table}" table in the {dataset} BigQuery Dataset

#'
#' @format A Google BigQuery Dataset table with {nrow} rows and {nvar} variables:
#' \describe{
{items}
#'   ...
#' }
#'

  tbl(src = project_connect(), "{dataset}.{table}")
}
